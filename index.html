async function cargarStock() { const { data, error } = await client.from('estudio_data').select('*').order('created_at', {ascending: false}); if(data) { lanasCargadas = data; // 1. Agrupamos los materiales por su categoría const grupos = data.reduce((acc, item) => { const cat = item.categoria || "Otros"; if (!acc[cat]) acc[cat] = []; acc[cat].push(item); return acc; }, {}); // 2. Emojis exactos recuperados const emojisCat = { "Telas": "👗", "Otros": "🧸", "Hilos": "🧵", "Lanas": "🧶", "Accesorios": "🎀" }; // 3. Generamos el HTML con separadores de subsección let htmlFinal = ""; // Ordenamos las categorías para que siempre salgan igual const ordenCategorias = ["Lanas", "Hilos", "Telas", "Accesorios", "Otros"]; // Primero las categorías que conocemos en orden ordenCategorias.forEach(cat => { if (grupos[cat]) { htmlFinal += generarBloqueCategoria(cat, emojisCat[cat], grupos[cat]); delete grupos[cat]; // Para no repetir } }); // Si queda alguna categoría extra que no esté en la lista for (const cat in grupos) { htmlFinal += generarBloqueCategoria(cat, "✨", grupos[cat]); } document.getElementById('lista-stock').innerHTML = htmlFinal; } } // Función auxiliar para mantener el código limpio y no tocar el diseño function generarBloqueCategoria(nombre, emoji, items) { let html = `

${nombre} ${emoji}

`; html += items.map(m => `
${m.contenido ? `` : '🧶'}

${m.tipo_material}

${m.composicion || ''} ${m.grosor ? '· '+m.grosor : ''}

${m.cantidad} uds
`).join(''); return html; }
